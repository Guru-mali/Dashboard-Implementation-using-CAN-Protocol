  ///////DTOH///////
  u8 dtoh(u8 n){
 	u8 t=0;
 	t=(n/10)<<4;
 	t|=(n%10);
 	return t;
 	}
  ///////SEC///////
  void set_sec(){
 
  u8 sec=0;
  lcd_cmd(0x80);
  lcd_string("SEC");
  while(1){
 	if(SW1 == 0){
 	 while(SW1 ==0);
 		  sec++;
 		  if(sec>=60)
 		  sec=0;
 		  }
 		 
 		 lcd_cmd(0xc0);
 		 if(sec<10)
 		 lcd_data('0');
 		 lcd_int(sec);
 		 
 	if(SW2==0){
 	while(SW2==0);
 	break;
 	 	 }
 	}
 	sec=dtoh(sec);
    i2c_byt_wrt_frm(0xd0,0x0,sec);
  }

  ////////////HR////////////
  void set_hr(){
  u8 hr,apm=0,mode;
  mode=set_mode();
 if(mode){
  apm=set_apm();
  }
   hr=set_time(mode);
   mode=mode<<6;
   apm=apm<<5;
   hr=hr|apm|mode;
   
   i2c_byt_wrt_frm(0xd0,0x2,hr);
  }
 
  u8 set_time(u8 mode){
  	 u8 hr=1;
  while(1){
 	if(SW1 == 0){
 	 while(SW1 ==0);
 		  hr++;
 		  if(mode){
 		  if(hr>=13)
 		  hr=1;
 		  }else{
 		  	if(hr>=24)
 		  hr=0;
 		  }
 		  }
 		 lcd_cmd(0x80);
 	     lcd_string("HR");
 		 lcd_cmd(0xc0);
 		 if(hr<10)
 		 lcd_data('0');
 		 lcd_int(hr);
 		 
 	if(SW2==0){
 	while(SW2==0);
 	break;
 	 	 }
 	}
 	lcd_cmd(0x01);
 	hr=dtoh(hr);
 	return hr;
    
  }
 
  u8 set_mode(){
 	   u8 zone= (i2c_byt_red_frm(0xd0,0x2)>>6)&1;
 	   lcd_cmd(0x80);
        lcd_string("MODE");
 	   while(1){
 	   if(SW1==0){
 	   while(SW1==0);
 	   zone^=1;
 	   }
 	   lcd_cmd(0xc0);
 	   if(zone){
 		  lcd_string("12 hr mode");
 		  }
 		  else{
 		  lcd_string("24 hr mode");
 		  }
 		if(SW2==0){
 		while(SW2==0);
 		break;
 		}
 	}
 		lcd_cmd(0x01);
 	return zone;
 }
 u8 set_apm(){
 	   u8 apm= (i2c_byt_red_frm(0xd0,0x2)>>5)&1;
 	   lcd_cmd(0x80);

 		while(SW2==0);
 		break;
 		}
 	}
 		lcd_cmd(0x01);
 	return apm;
 }

 	    }
 	  }

 	     lcd_string("YEARS");
 		 lcd_cmd(0xc0);

 		 lcd_data('0');

 		 	if(SW2==0){

              	break;
 	 	 }
 
 	}

 	years=dtoh(years);
    i2c_byt_wrt_frm(0xd0,0x6,years);
 }

   ///////////////////DAY
   void set_day(){
  
    u8 day=1;
     lcd_cmd(0x01);
  while(1){
 	if(SW1 == 0){
 	 while(SW1 ==0);
 		  day++;
 		  if(day>=8)
 		  day=1;
 		  }
 		 lcd_cmd(0x80);
 	     lcd_string("DAY");
 		 lcd_cmd(0xc0);
 		switch(day){
 		 case 1: lcd_string("SUN");break;
 		 case 2: lcd_string("MON");break;
 		 case 3: lcd_string("TUE");break;
 		 case 4: lcd_string("WED");break;
 		 case 5: lcd_string("THU");break;
 		 case 6: lcd_string("FRI");break;
 		 case 7: lcd_string("SAT");break;
 		}
 		 
 	if(SW2==0){
 	while(SW2==0);
 	break;
 	 	 }
 	}
    i2c_byt_wrt_frm(0xd0,0x3,day);
  }
